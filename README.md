```bash
# find install path
where node
where npm

# 檢視環境變數 path
set path

# 設定安裝路徑
npm config list
npm config get prefix
npm config set prefix

npm install npm --global
npm list --global
npm list -g
npm uninstall "套件"
npm update "套件"
npm search "套件"
npm cache clean

npm init
npm init --yes
npm init -y

# npm cache clean
npm cache verify

node
process.memoryUsage()

# node red install
npm install -g --unsafe-perm node-red

cd C:\Users\Admin\.node-red
npm install node-red-node-arduino
```
